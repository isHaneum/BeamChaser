7. 데이터 구조 (핵심)

이거 없으면 무조건 느려진다.

type RunRecord = {
  id: string
  date: string // "2026-04-28"
  distance: number
  duration: number
  pace: number
}

type CalendarMap = {
  [date: string]: {
    count: number
    totalDistance: number
  }
}
8. 데이터 생성 로직
function buildCalendarMap(records: RunRecord[]): CalendarMap {
  const map: CalendarMap = {}

  for (const r of records) {
    if (!map[r.date]) {
      map[r.date] = { count: 0, totalDistance: 0 }
    }

    map[r.date].count += 1
    map[r.date].totalDistance += r.distance
  }

  return map
}
9. 달력 셀 렌더링
function renderDay(date) {
  const data = calendarMap[date]

  return {
    isRun: !!data,
    isMulti: data?.count > 1,
    isToday: date === today,
    isSelected: date === selectedDate
  }
}
10. 리스트 연동 (중요)
function getFilteredRuns(records, selectedDate) {
  if (!selectedDate) return records

  return records.filter(r => r.date === selectedDate)
}
